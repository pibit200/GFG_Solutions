Problem Statement: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------

Queue Reversal

-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
Solution: 
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------
class Solution
{
    public:
    queue<int> rev(queue<int> q)
    {
        // add code here.
        queue <int> Q;
        stack <int> S;
        
        while(!q.empty()){
            S.push(q.front());
            q.pop();
        }
        while(!S.empty()){
            Q.push(S.top());
            S.pop();
        }
        return Q;
    }
};
