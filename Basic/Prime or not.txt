class Solution{
public:
    int isPrime(int N){
        // code here
       if(N <= 1){
           return false;
       }
       else{
           for(int i=2; i<N/2; i++){
               if((N%i) == 0){
                   return false;
               }
           }
           return true;
       }
    }
};


//{ Driver Code Starts.
int main() 
{ 
    int t;
    cin>>t;
    while(t--)
    {
        int N;
        cin>>N;
        Solution ob;
        cout << ob.isPrime(N) << endl;
    }
    return 0; 
}
